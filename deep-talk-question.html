    <script>

        // *** IMPORTANT: This CSV URL points to your Google Sheet. ***

        // Please ensure the desired tab is published as CSV.

        const GOOGLE_SHEET_CSV_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSNhi4Yrxv3hdVKz-OD2kW4iDP-rQe5vJR13XqKFlIZhVu3HKPkDKHDtTci2CpzS1UjSpLusehbfX1s/pub?gid=2122510469&single=true&output=csv'; 


        const randomizeButton = document.getElementById('randomizeButton');

        const questionDisplay = document.getElementById('questionDisplay');

        const loadingIndicator = document.getElementById('loadingIndicator');

        const errorMessage = document.getElementById('errorMessage');


        let questions = []; // Array to store questions from the Google Sheet


        // Function to show loading state

        function showLoading() {

            loadingIndicator.classList.remove('hidden');

            randomizeButton.disabled = true;

            questionDisplay.innerHTML = '<p class="text-gray-500">Loading questions...</p>';

            errorMessage.classList.add('hidden');

        }


        // Function to hide loading state

        function hideLoading() {

            loadingIndicator.classList.add('hidden');

            randomizeButton.disabled = false;

        }


        // Function to show error message

        function showError(message) {

            errorMessage.textContent = message;

            errorMessage.classList.remove('hidden');

            questionDisplay.innerHTML = '<p class="text-gray-500">An error occurred</p>';

        }


        // Function to fetch and parse CSV data

        async function fetchQuestions() {

            showLoading();

            try {

                const response = await fetch(GOOGLE_SHEET_CSV_URL);

                if (!response.ok) {

                    throw new Error(`Failed to load data: ${response.statusText}`);

                }

                const csvText = await response.text();

                

                // Parse CSV: Split by new line, then take only the first column (index 0)

                // This ensures only questions from Column A are used.

                questions = csvText.split('\n')

                                   .map(line => line.split(',')[0].trim()) // Get the first column and trim whitespace

                                   .filter(line => line.length > 0); // Filter out any empty lines/questions


                if (questions.length === 0) {

                    showError('No questions found.');

                } else {

                    // Display initial prompt or first question if desired

                    questionDisplay.innerHTML = '<p class="text-gray-500">Ready to share your story?</p>';

                }

            } catch (error) {

                console.error('Error fetching questions:', error);

                showError(`Could not load questions: ${error.message}`);

            } finally {

                hideLoading();

            }

        }


        // Function to get a random question

        function getRandomQuestion() {

            if (questions.length === 0) {

                questionDisplay.innerHTML = '<p class="text-gray-500">No questions available.</p>';

                return;

            }

            const randomIndex = Math.floor(Math.random() * questions.length);

            questionDisplay.textContent = questions[randomIndex];

        }


        // Event listener for the button

        randomizeButton.addEventListener('click', getRandomQuestion);


        // Fetch questions when the page loads

        window.onload = fetchQuestions;
